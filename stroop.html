function doPost(e) { try { // 送信されてきたJSONデータをパース const jsonData = JSON.parse(e.postData.contents); // 書き込み先のスプレッドシートを取得（アクティブなシート、または名前で指定） const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getActiveSheet(); // データの書き込み jsonData.forEach(function(row) { // index.htmlのmap関数で定義したプロパティ名と一致させる sheet.appendRow([ row.subject_id, row.timestamp, row.word, row.color, row.condition, row.rt, row.correct ]); }); return ContentService.createTextOutput("Success"); } catch (error) { return ContentService.createTextOutput("Error: " + error.toString()); } }
